IGNORE : "[ \f\t\n\r]+";

NUMBERLITERAL: "(((0|[1-9][0-9]*)(\.[0-9]*)?)|(\.[0-9]+))";
IDENTIFIERNAME: "[a-zA-Z_][a-zA-Z0-9_]*";

main : sourceelements [EOF];

sourceelements : sourceelement >sourceelements<
               | sourceelement
               ;

sourceelement : <statement>
              ;

statement : <additiveexpression>
          | <assignmentexpression>
          ;

muloperator : "*" | "/";
multiplicativeexpression : number >muloperator< multiplicativeexpression
                         | number
                         ;

addoperator : "+"| "-";
additiveexpression : multiplicativeexpression >addoperator< additiveexpression
                   | multiplicativeexpression
                   ;

assignmentoperator : "<-"
                   ;
assignmentexpression : IDENTIFIERNAME >assignmentoperator< number
                     ;

number : NUMBERLITERAL 
       ;
